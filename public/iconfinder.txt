/* fa fa-check-square-o (done) */

html:  <i class={{status}}></i>
js:  
Template.AccActivityItem.helpers({
  status: function() {
    var activity = Activities.findOne(this._id);
    var teacher = Roles.userIsInRole(Meteor.userId(),'teacher') ? ' teacher' : '';
    if (!activity) return ''; 
    if (activity.status == 'submitted') return 'icon-submitted' + teacher;
    if (activity.status == 'returned') return 'icon-returned' + teacher;
    if (activity.status == 'done') return 'fa fa-check-square-o';
    return 'fa fa-square-o';    
  }
});
